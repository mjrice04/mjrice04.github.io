# Image handling

def get_image(df_url):    
    a_id = df_url['albumId']
    image_id = df_url['albumarturl']
    img_data = requests.get(image_id).content
    with open(f'/home/matt/projects/google_music_analytics/data/images/{a_id}.jpg', 'wb') as handler:
        handler.write(img_data)
        
pd.set_option('display.max_colwidth', -1)

def get_thumbnail(path):
    i = Image.open(path)
    i.thumbnail((150, 150), Image.LANCZOS)
    return i

def image_base64(im):
    if isinstance(im, str):
        im = get_thumbnail(im)
    with BytesIO() as buffer:
        im.save(buffer, 'jpeg')
        return base64.b64encode(buffer.getvalue()).decode()

def image_formatter(im):
    return f'<img src="data:image/jpeg;base64,{image_base64(im)}">'

def get_top_albums(albums, year=None, years=None):
    """
    returns Dataframe with top albums. year can be passed in to get albums from a year.
    years can be passed in to get a range of years
    :param albums: albums dataframe
    :param year: year to get top albums for
    :param years: range of years to get top albums for
    :return:
    """
    album_subset = albums[['album', 'artist', 'playCount', 'albumId', 'year']]
    album_subset_year = album_subset.loc[album_subset['year'] == year]
    clean_album_subset = album_subset_year[['album', 'artist', 'playCount', 'albumId']]
    grouped_subset = clean_album_subset.groupby(['album', 'artist', 'albumId'], as_index=False)[['playCount']].sum()
    top_albums = grouped_subset.sort_values(by=['playCount'], ascending=False)
    ta = top_albums.head(10)
    return ta

def merge_in_id(df1, df2, col):
    df = pd.merge(df1, df2, how='inner', on=col)
    return df

def image_frame(songs, clean_album_art, year):
    top_album_frame = get_top_albums(songs, year)
    formatted_top_albums = merge_in_id(top_album_frame, clean_album_art, 'albumId')
    formatted_top_albums.apply(get_image, axis = 1)
    formatted_top_albums_image = formatted_top_albums.copy()
    formatted_top_albums_image['file'] = formatted_top_albums_image.albumId.map(lambda id: f'/home/matt/projects/google_music_analytics/data/images/{id}.jpg')
    formatted_top_albums_image['image'] = formatted_top_albums_image.file.map(lambda f: get_thumbnail(f))
    return formatted_top_albums_image
